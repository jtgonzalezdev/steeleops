import os
import sqlite3
from flask import Flask, request, redirect, url_for, jsonify, render_template_string

app = Flask(__name__)

DATABASE_URL = os.getenv("DATABASE_URL", "steeleops.db")


def get_db():
    if DATABASE_URL.startswith("postgres"):
        import psycopg2
        from psycopg2.extras import RealDictCursor
        conn = psycopg2.connect(DATABASE_URL)
        conn.autocommit = True
        return conn
    else:
        conn = sqlite3.connect(DATABASE_URL)
        conn.row_factory = sqlite3.Row
        return conn


def get_cursor(conn):
    if DATABASE_URL.startswith("postgres"):
        from psycopg2.extras import RealDictCursor
        return conn.cursor(cursor_factory=RealDictCursor)
    return conn.cursor()


def fetch_count(cur):
    row = cur.fetchone()
    if not row:
        return 0
    if isinstance(row, dict):
        return list(row.values())[0]
    if hasattr(row, "keys"):
        keys = row.keys()
        return row[keys[0]]
    return row[0]


def row_to_dict(row):
    if isinstance(row, dict):
        return row
    return dict(row)


def init_db():
    conn = get_db()
    cur = get_cursor(conn)

    if DATABASE_URL.startswith("postgres"):
        cur.execute("""
            CREATE TABLE IF NOT EXISTS companies (
                id SERIAL PRIMARY KEY,
                name TEXT NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        cur.execute("""
            CREATE TABLE IF NOT EXISTS guards (
                id SERIAL PRIMARY KEY,
                name TEXT NOT NULL,
                company_id INTEGER,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        cur.execute("""
            CREATE TABLE IF NOT EXISTS shifts (
                id SERIAL PRIMARY KEY,
                guard_id INTEGER,
                start_time TIMESTAMP,
                end_time TIMESTAMP,
                location TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
    else:
        cur.execute("""
            CREATE TABLE IF NOT EXISTS companies (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        cur.execute("""
            CREATE TABLE IF NOT EXISTS guards (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                company_id INTEGER,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        cur.execute("""
            CREATE TABLE IF NOT EXISTS shifts (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                guard_id INTEGER,
                start_time TIMESTAMP,
                end_time TIMESTAMP,
                location TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)

    cur.execute("SELECT COUNT(*) AS cnt FROM companies")
    company_count = fetch_count(cur)

    if company_count == 0:
        if DATABASE_URL.startswith("postgres"):
            cur.execute(
                "INSERT INTO companies (name) VALUES (%s)",
                ("Steele Security Services",)
            )
        else:
            cur.execute(
                "INSERT INTO companies (name) VALUES (?)",
                ("Steele Security Services",)
            )

    conn.commit()
    conn.close()

@app.route("/api/status")
def api_status():
    conn = get_db()
    cur = get_cursor(conn)

    cur.execute("SELECT COUNT(*) AS cnt FROM guards")
    guards = fetch_count(cur)

    cur.execute("SELECT COUNT(*) AS cnt FROM shifts")
    shifts = fetch_count(cur)

    conn.close()

    return jsonify({
        "platform": "SteeleOps",
        "guards": guards,
        "shifts": shifts,
        "status": "running"
    })


@app.route("/guards", methods=["GET", "POST"])
def manage_guards():
    conn = get_db()
    cur = get_cursor(conn)

    if request.method == "POST":
        name = request.form.get("name")
        company_id = request.form.get("company_id")

        if DATABASE_URL.startswith("postgres"):
            cur.execute(
                "INSERT INTO guards (name, company_id) VALUES (%s, %s)",
                (name, company_id)
            )
        else:
            cur.execute(
                "INSERT INTO guards (name, company_id) VALUES (?, ?)",
                (name, company_id)
            )

        conn.commit()
        conn.close()
        return redirect(url_for("dashboard"))

    cur.execute("SELECT * FROM guards ORDER BY id DESC")
    guards = [row_to_dict(row) for row in cur.fetchall()]
    conn.close()
    return jsonify(guards)


@app.route("/shifts", methods=["GET", "POST"])
def manage_shifts():
    conn = get_db()
    cur = get_cursor(conn)

    if request.method == "POST":
        guard_id = request.form.get("guard_id")
        start_time = request.form.get("start_time")
        end_time = request.form.get("end_time")
        location = request.form.get("location")

        if DATABASE_URL.startswith("postgres"):
            cur.execute(
                "INSERT INTO shifts (guard_id, start_time, end_time, location) VALUES (%s, %s, %s, %s)",
                (guard_id, start_time, end_time, location)
            )
        else:
            cur.execute(
                "INSERT INTO shifts (guard_id, start_time, end_time, location) VALUES (?, ?, ?, ?)",
                (guard_id, start_time, end_time, location)
            )

        conn.commit()
        conn.close()
        return redirect(url_for("dashboard"))

    cur.execute("SELECT * FROM shifts ORDER BY id DESC")
    shifts = [row_to_dict(row) for row in cur.fetchall()]
    conn.close()
    return jsonify(shifts)
